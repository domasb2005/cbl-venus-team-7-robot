33 typedef enum _led_mode { uninitialized, binary, pwm_green, pwm_color } led_mode;

36 // LEDs are either on or off
37 void leds_init_onoff(void) {
38  if (mode == binary)
39  return;
40  if (mode != uninitialized) {
41  pynq_error("leds_init_onoff: mode=%d should be uninitialized\n", mode);
42  }
43  gpio_set_direction(IO_LD0, GPIO_DIR_OUTPUT);
44  gpio_set_direction(IO_LD1, GPIO_DIR_OUTPUT);
45  gpio_set_direction(IO_LD2, GPIO_DIR_OUTPUT);
46  gpio_set_direction(IO_LD3, GPIO_DIR_OUTPUT);
47  gpio_set_direction(IO_LD4B, GPIO_DIR_OUTPUT);
48  gpio_set_direction(IO_LD4G, GPIO_DIR_OUTPUT);
49  gpio_set_direction(IO_LD4R, GPIO_DIR_OUTPUT);
50  gpio_set_direction(IO_LD5B, GPIO_DIR_OUTPUT);
51  gpio_set_direction(IO_LD5G, GPIO_DIR_OUTPUT);
52  gpio_set_direction(IO_LD5R, GPIO_DIR_OUTPUT);
53  mode = binary;
54 }

56 // can change the intensity of LEDs, the onoff parameters are then in the range
57 // 0..255
58 void green_leds_init_pwm(void) {
59  if (mode == pwm_green)
60  return;
61  if (mode != uninitialized) {
62  pynq_error("green_leds_init_pwm: mode=%d should be uninitialized\n", mode);
63  }
64  // initialize switchbox and routing PWM to LEDs
65  switchbox_set_pin(IO_LD0, SWB_PWM0);
66  switchbox_set_pin(IO_LD1, SWB_PWM1);
67  switchbox_set_pin(IO_LD2, SWB_PWM2);
68  switchbox_set_pin(IO_LD3, SWB_PWM3);
69  // initialize the PWM channels
70  pwm_init(PWM0, 256);
71  pwm_init(PWM1, 256);
72  pwm_init(PWM2, 256);
73  pwm_init(PWM3, 256);
74  mode = pwm_green;
75 }

96 void leds_destroy(void) {
97  // note that pynq_destroy will also reset all GPIO and switch off all LEDs
98  if (mode == binary) {
99  for (int i = 0; i < NUM_GREEN_LEDS; i++)
100  green_led_off(i);
101  }
102  if (mode == pwm_green || mode == pwm_color) {
103  green_led_off(0);
104  green_led_off(1);
105  green_led_off(2);
106  pwm_destroy(PWM0);
107  pwm_destroy(PWM1);
108  pwm_destroy(PWM2);
109  }
110  if (mode == pwm_green) {
111  green_led_off(3);
112  pwm_destroy(PWM3);
113  }
114  mode = uninitialized;
115 }

117 void green_led_onoff(const int led, const int onoff) {
118  if (led < 0 || led >= NUM_GREEN_LEDS) {
119  pynq_error("green_led_onoff: invalid led=%d, must be 0..%d-1\n",
120  NUM_GREEN_LEDS);
121  }
122  int oo = onoff;
123  switch (mode) {
124  case binary:
125  gpio_set_level(IO_LD0 + led,
126  (onoff == LED_OFF ? GPIO_LEVEL_LOW : GPIO_LEVEL_HIGH));
127  break;
128  case pwm_green:
129  case pwm_color:
130  if (onoff < 0) {
131  oo = 0;
132  } else {
133  if (onoff > 255) {
134  oo = 255;
135  }
136  }
137  pwm_set_duty_cycle(PWM0 + led, oo);
138  break;
139  default:
140  pynq_error("green_led_onoff: LEDs have not been initialized with "
141  "green_leds_init_pwm\n");
142  break;
143  }
144 }
